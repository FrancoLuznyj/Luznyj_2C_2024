110static void medirDistancia(void *pvParameter)
111{
112 uint16_t numero_muestra=0;
113 uint16_t distancia_nueva=0;
114
115 while(true)
116 {
117 distancia_nueva = HcSr04ReadDistanceInCentimeters();
118
119 for(uint8_t i=1;i<BUFFER_SIZE_DISTANCIAS;i++)
120 {
121 vector_distancias[i-1] = vector_distancias[i];
122 }
123 vector_distancias[BUFFER_SIZE_DISTANCIAS] = distancia_nueva;
124
125 if(distancia_nueva < 1000)
126 {
127 velocidad = ((vector_distancias[BUFFER_SIZE_DISTANCIAS] - vector_distancias[BUFFER_SIZE_DISTANCIAS-1]) / 100) / DELTA_T;
128
129 if(velocidad_maxima < velocidad)
130 {
131 velocidad_maxima = velocidad;
132 }
133
134 if(velocidad >= 8)
135 {
136 LedOn(LED_3);
137 LedOff(LED_2);
138 LedOff(LED_1);
139 }
140 else if(velocidad > 0 && velocidad < 8)
141 {
142 LedOn(LED_2);
143 LedOff(LED_3);
144 LedOff(LED_1);
145 }
146 else
147 {
148 LedOn(LED_1);
149 LedOff(LED_2);
150 LedOff(LED_3);
151 }
152 }
153
154 vTaskDelay(CONFIG_BLINK_PERIOD_MEDIR_DISTANCIA_MS / portTICK_PERIOD_MS);
155 }

158static void PesarVehiculo(void *pvParameter)
159{
160 uint16_t galga_1 = 0;
161 uint16_t galga_2 = 0;
162 uint16_t mediciones_para_promedio = 0;
163 uint16_t promedio_galga_1 = 0;
164 uint16_t promedio_galga_2 = 0;
165
166 while(true)
167 {
168 AnalogInputReadSingle(CH1, &galga_1);
169 AnalogInputReadSingle(CH2, &galga_2);
170
171 galga_1 = (0 + ( ( (20000-0) / (3,3 - 0) ) * (galga_1 - 0) ));
172 galga_2 = (0 + ( ( (20000-0) / (3,3 - 0) ) * (galga_2 - 0) ));
173
174 vector_pesos_galga1[mediciones_para_promedio] = galga_1;
175 vector_pesos_galga2[mediciones_para_promedio] = galga_2;
176
177 mediciones_para_promedio++;
178
179 if(mediciones_para_promedio == BUFFER_SIZE_PESOS-1)
180 {
181 promedio_galga_1 = vector_pesos_galga1[0];
182 promedio_galga_2 = vector_pesos_galga2[0];
183
184 for(uint8_t i=0;i<BUFFER_SIZE_PESOS;i++)
185 {
186 promedio_galga_1 = promedio_galga_1 + vector_pesos_galga1[i];
187 promedio_galga_2 = promedio_galga_2 + vector_pesos_galga2[i];
188 }
189
190 promedio_galga_1 = promedio_galga_1 / BUFFER_SIZE_PESOS;
191 promedio_galga_2 = promedio_galga_2 / BUFFER_SIZE_PESOS;
192
193 peso_vehiculo = promedio_galga_1 + promedio_galga_2;
194 }
195
196 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
197 }
