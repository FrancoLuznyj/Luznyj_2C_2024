97static void MedirDistancia(void *pvParameter)
98{
99 while(true)
100 {
101 distancia = HcSr04ReadDistanceInCentimeters();
102
103 if(distancia > 500)
104 {
105 LedOn(LED_1);
106 LedOff(LED_2);
107 LedOff(LED_3);
108 }
109
110 else if(distancia <= 500 && distancia >= 300)
111 {
112 LedOn(LED_1);
113 LedOn(LED_2);
114 LedOff(LED_3);

191static void MedirAceleracion(void *pvParameter)
192{
193 uint16_t aceleracion_x = 0, aceleracion_y = 0, aceleracion_z = 0;
194 while(true)
195 {
196 //Mido la aceleracion en "volts" que me da el acelerometro
197 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
198 AnalogInputReadSingle(CH1, &aceleracion_x);
199 AnalogInputReadSingle(CH2, &aceleracion_y);
200 AnalogInputReadSingle(CH3, &aceleracion_z);
201
202 //Paso de voltaje a valor de aceleracion en "G"
203 aceleracion_x = (aceleracion_x * 5,5) / 3,3;
204 aceleracion_y = (aceleracion_y * 5,5) / 3,3;
205 aceleracion_z = (aceleracion_z * 5,5) / 3,3;
206
207 aceleracion = aceleracion_x + aceleracion_y + aceleracion_z;
208 }
